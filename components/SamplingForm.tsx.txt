import React, { useState, useEffect } from 'react';
import { SampleParams, SampleResult } from '../types';
import { generateSystematicSample } from '../services/samplingService';

interface SamplingFormProps {
  onGenerate: (results: SampleResult) => void;
}

const SamplingForm: React.FC<SamplingFormProps> = ({ onGenerate }) => {
  const [grade, setGrade] = useState('6');
  const [totalStudents, setTotalStudents] = useState('');
  const [sampleSize, setSampleSize] = useState('');
  const [reserveSize, setReserveSize] = useState('3');
  const [error, setError] = useState('');

  useEffect(() => {
    const n = parseInt(sampleSize, 10);
    const N = parseInt(totalStudents, 10);
    if (!isNaN(n) && !isNaN(N) && n > N) {
      setError('Sample size cannot exceed total students.');
    } else {
      setError('');
    }
  }, [sampleSize, totalStudents]);

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    setError('');
    const N = parseInt(totalStudents, 10);
    const n = parseInt(sampleSize, 10);
    const r = parseInt(reserveSize, 10);

    if (isNaN(N) || N <= 0) {
      setError('Please enter a valid total number of students.');
      return;
    }
    if (isNaN(n) || n <= 0) {
      setError('Please enter a valid sample size.');
      return;
    }
    if (n > N) {
      setError('Sample size cannot exceed total students.');
      return;
    }
    if (isNaN(r) || r < 0) {
      setError('Please enter a valid reserve list size.');
      return;
    }

    try {
      const params: SampleParams = {
        grade,
        totalStudents: N,
        sampleSize: n,
        reserveSize: r
      };
      const results = generateSystematicSample(params);
      onGenerate(results);
    } catch (err: any) {
      setError(err.message);
    }
  };

  const inputClasses = "mt-1 block w-full px-3 py-2 bg-white border border-slate-300 rounded-md text-sm shadow-sm placeholder-slate-400 focus:outline-none focus:border-sky-500 focus:ring-1 focus:ring-sky-500";

  return (
    <div className="bg-white rounded-xl shadow-lg p-6 sm:p-8">
      <h2 className="text-2xl font-bold text-slate-700 mb-6 text-center">Sampling Details</h2>
      <form onSubmit={handleSubmit} className="space-y-6">
        <div>
          <label htmlFor="grade" className="block text-sm font-medium text-slate-700">Grade</label>
          <select id="grade" value={grade} onChange={e => setGrade(e.target.value)} className={inputClasses}>
            <option value="6">Grade 6</option>
            <option value="7">Grade 7</option>
            <option value="8">Grade 8</option>
          </select>
        </div>
        <div>
          <label htmlFor="total-students" className="block text-sm font-medium text-slate-700">Total students present (N)</label>
          <input type="number" id="total-students" value={totalStudents} onChange={e => setTotalStudents(e.target.value)} className={inputClasses} placeholder="e.g., 85" min="1" required />
        </div>
        <div>
          <label htmlFor="sample-size" className="block text-sm font-medium text-slate-700">Number of students to sample (n)</label>
          <input type="number" id="sample-size" value={sampleSize} onChange={e => setSampleSize(e.target.value)} className={inputClasses} placeholder="e.g., 15" min="1" required />
        </div>
        <div>
          <label htmlFor="reserve-size" className="block text-sm font-medium text-slate-700">Reserve list size</label>
          <input type="number" id="reserve-size" value={reserveSize} onChange={e => setReserveSize(e.target.value)} className={inputClasses} placeholder="Default: 3" min="0" required />
        </div>
        
        {error && <p className="text-sm text-red-600 bg-red-100 p-3 rounded-md">{error}</p>}

        <button type="submit" className="w-full bg-blue-600 text-white font-bold py-3 px-4 rounded-lg hover:bg-blue-700 focus:outline-none focus:ring-2 focus:ring-blue-500 focus:ring-opacity-50 transition-transform transform hover:scale-105 disabled:bg-slate-400" disabled={!!error || !totalStudents || !sampleSize}>
          Generate Sample
        </button>
      </form>
    </div>
  );
};

export default SamplingForm;
